インストール

yarn

使い方

クラスリファレンス

<!DOCTYPE html>
<html lang="js">
<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Static Sample</title>
    <script src="/usbif4cw.js"></script>
    <script>
        const onClick = async () => {
            const device = await usbif4cw.requestDevice();
            await device.open();
            await device.outputCWUsingString("HELLO");
        }
    </script>
</head>
<body>
    <button onclick="onClick()">接続</button>
</body>
</html>
